[A, B, pos_var, ~] = small_sgn_linp('Grid_WSCC_Sauer','gendat_WSCC_9_bus',  1,  1);

%% Eigenanalysis
[Righteigv,omega,Lefteigv]=eig(A); %Getting Right and left eigenvectors
eigenvalues=eig(A); %Writing eigenvalue in a Mx1 matrix
alpha=real(eigenvalues); %Real part of the eigenvalue is alpha
omega_d=imag(eigenvalues); %Imaginary part of the eigenvalue is omega_d (damped frequency)
frequency_d=omega_d/(2*pi); %Frequency of the oscillation
damping_ratio =-alpha./abs(omega);%Formula obtained by playing w/ eq. in slide 11 pg 15
damping_ratio=diag(damping_ratio); %Writing it as a vector Mx1

%% Study of the right eigenvectors
Reigen_study=diag(Righteigv); %Study of the diagonal elements to consider what modes are excited
Reigen_study=round(Reigen_study,6); %Rounding 
%If the diagonal element is zero -> Mode not excited (I think)

%% Participation factor
%To study the participation factor only the magnitude is required (hence
%the abs(), THE FOLLOWING ONE IS NOT THE PARTICIPATION MATRIX, IT IS THE
%MAGNITUDES OF THE PARTICIPATION MATRIX.
part_matrix=abs(Righteigv.*Lefteigv); %Definition of participation factor (See the exctract in shared folder pg.14)
%L1 Norm
L1_part_matrix=sum(abs(part_matrix)); %L1 norm
part_matrix=part_matrix./L1_part_matrix; %Normalization

Real_part_matrix=Lefteigv.*(Righteigv).';%Complex participation matrix
Real_part_matrix=Real_part_matrix./abs(max(Real_part_matrix)); %Normalization
